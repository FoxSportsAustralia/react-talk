fscsw.components.matchtracker = React.createClass({displayName: 'Match Tracker',
    render: function() {
        return (
            React.DOM.div(null,
                React.DOM.div(
                    null,
                    fscsw.components.tabs({
                        activeTab: self.props.activeTab,
                        tabs: this.props.tabs,
                        onClick: function(clickedId) {
                            self.props.onTabChange(clickedId);
                        }
                    })
                ),
                React.DOM.div(
                    {className: 'fscsw-tracker-border'},
                    React.DOM.div(
                        null,
                        React.DOM.div(
                            {className: classesFormation},
                            fscsw.components.matchtrackerFormation({
                                homeTeamId    : this.props.homeTeamId,
                                homeFormation : this.props.formation.homeFormation,
                                awayTeamId    : this.props.awayTeamId,
                                awayFormation : this.props.formation.awayFormation,

                                fieldImage : this.props.fieldImage,

                                selectedPlayerId      : this.props.formationSelectedPlayer,
                                onFormationClick      : this.props.onFormationClick,
                                onFormationMouseEnter : this.props.onFormationMouseEnter,
                                onFormationMouseLeave : this.props.onFormationMouseLeave
                            })
                        ),
                        React.DOM.div(
                            {className: classesEvents},
                            fscsw.components.matchtrackerEvents({
                                fieldImage   : this.props.fieldImage,
                                filterEvents : this.props.filterEvents,
                                events       : this.props.events
                            })
                        )
                    )
                ),
                React.DOM.div(
                    {className: 'fscsw-possession-border'},
                    React.DOM.div(null, fscsw.components.matchtrackerPossession({possession: this.props.possession}))
                )
            )

        );
    }
});
